52  /**
53   * <p>This class provides a simplified API similar to {@link HTTPBuilder}, but 
54   * uses {@link java.net.HttpURLConnection} for I/O so that it is compatible 
55   * with Google App Engine.  Features:
56   * <ul>
57   *  <li>Parser and Encoder support</li>
58   *  <li>Easy request and response header manipulation</li>
59   *  <li>Basic authentication</li>
60   * </ul>
61   * Notably absent are status-code based response handling and the more complex 
62   * authentication mechanisms.</p>
63   * 
64   * TODO request encoding support?
65   *
66   * @see <a href='http://code.google.com/appengine/docs/java/urlfetch/overview.html'>GAE URLFetch</a>
67   * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
68   * @since 0.5.0
69   */

83  	/**
84  	 * Perform a request.  Parameters are:
85  	 * <dl>
86  	 *   <dt>url</dt><dd>the entire request URL</dd>
87  	 *   <dt>path</dt><dd>the path portion of the request URL, if a default
88  	 *     URL is set on this instance.</dd>
89  	 *   <dt>query</dt><dd>URL query parameters for this request.</dd>
90  	 *   <dt>timeout</dt><dd>see {@link HttpURLConnection#setReadTimeout(int)}</dd>
91  	 *   <dt>method</dt><dd>This defaults to GET, or POST if a <code>body</code> 
92  	 *   parameter is also specified.</dd>
93  	 *   <dt>contentType</dt><dd>Explicitly specify how to parse the response.
94  	 *     If this value is ContentType.ANY, the response <code>Content-Type</code>
95  	 *     header is used to determine how to parse the response.</dd>
96  	 *   <dt>requestContentType</dt><dd>used in a PUT or POST request to 
97  	 *     transform the request body and set the proper 
98  	 *     <code>Content-Type</code> header.  This defaults to the 
99  	 *     <code>contentType</code> if unset.</dd>
100 	 *   <dt>auth</dt><dd>Basic authorization; pass the value as a list in the 
101 	 *   form [user, pass]</dd>
102 	 *   <dt>headers</dt><dd>additional request headers, as a map</dd>
103 	 *   <dt>body</dt><dd>request content body, for a PUT or POST request.  
104 	 *     This will be encoded using the requestContentType</dd>
105 	 * </dl>
106 	 * @param args named parameters
107 	 * @return the parsed response
108 	 * @throws URISyntaxException
109 	 * @throws MalformedURLException
110 	 * @throws IOException
111 	 */
112 	public HttpResponseDecorator request( Map<String,?> args ) 
113 			throws URISyntaxException, MalformedURLException, IOException {
114 		
115 		// copy so we don't modify the original collection when removing items:
116 		args = new HashMap<String,Object>(args);
117 		
118 		Object arg = args.remove( "url" );
119 		if ( arg == null && this.defaultURL == null )
120 			throw new IllegalStateException( "Either the 'defaultURL' property" +
121 					" must be set or a 'url' parameter must be passed to the " +
122 					"request method." );
123 		URIBuilder url = arg != null ? new URIBuilder( arg.toString() ) : defaultURL.clone();
124 
125 		arg = null;
126 		arg = args.remove( "path" );
127 		if ( arg != null ) url.setPath( arg.toString() );
128 		arg = null;
129 		arg = args.remove( "query" );
130 		if ( arg != null ) {
131 			if ( ! ( arg instanceof Map ) ) 
132 				throw new IllegalArgumentException( "'query' must be a map" ); 
133 			url.setQuery( (Map<?,?>)arg );
134 		}
135 		
136 		HttpURLConnection conn = (HttpURLConnection)url.toURL().openConnection();
137 		conn.setInstanceFollowRedirects( this.followRedirects );
138 		
139 		arg = null;
140 		arg = args.remove( "timeout" );
141 		if ( arg != null ) 
142 			conn.setConnectTimeout( Integer.parseInt( arg.toString() ) );
143 		
144 		arg = null;
145 		arg = args.remove( "method" );
146 		if ( arg != null ) conn.setRequestMethod( arg.toString() );
147 
148 		arg = null;
149 		arg = args.remove( "contentType" );
150 		Object contentType = arg != null ? arg : this.contentType;
151 		if ( contentType instanceof ContentType ) conn.addRequestProperty( 
152 				"Accept", ((ContentType)contentType).getAcceptHeader() );
153 		
154 		arg = null;
155 		arg = args.remove( "requestContentType" );
156 		String requestContentType = arg != null ? arg.toString() : 	
157 				this.requestContentType != null ? this.requestContentType.toString() : 
158 					contentType != null ? contentType.toString() : null;
159 					
160 		// must add default headers before setting auth:
161 		for ( String key : defaultHeaders.keySet() )
162 			conn.addRequestProperty( key, defaultHeaders.get( key ) );
163 		
164 		arg = null;
165 		arg = args.remove( "auth" );
166 		if ( arg != null ) {
167 			try { 
168 				List<?> vals = (List<?>)arg;
169 				conn.addRequestProperty( "Authorization", getBasicAuthHeader( 
170 						vals.get(0).toString(), vals.get(1).toString() ) );
171 			} catch ( Exception ex ) {
172 				throw new IllegalArgumentException( 
173 						"Auth argument must be a list in the form [user,pass]" );
174 			}
175 		}
176 					
177 		arg = null;
178 		arg = args.remove( "headers" );
179 		if ( arg != null ) {
180 			if ( ! ( arg instanceof Map ) ) 
181 				throw new IllegalArgumentException( "'headers' must be a map" ); 
182 			Map<?,?> headers = (Map<?,?>)arg;
183 			for ( Object key : headers.keySet() ) conn.addRequestProperty( 
184 					key.toString(), headers.get( key ).toString() );
185 		}
186 		
187 		arg = null;
188 		arg = args.remove( "body" );
189 		if ( arg != null ) {
190 			conn.setDoOutput( true );
191 			HttpEntity body = (HttpEntity)encoderRegistry.getAt( 
192 					requestContentType ).call( arg );
193 			// TODO configurable request charset
194 			
195 			//TODO don't override if there is a 'content-type' in the headers list
196 			conn.addRequestProperty( "Content-Type", requestContentType );
197 			try {
198 				DefaultGroovyMethods.leftShift( conn.getOutputStream(), 
199 						body.getContent() );
200 			}
201 			finally { conn.getOutputStream().close(); }
202 		}
203 		if ( args.size() > 0 ) for ( Object k : args.keySet() ) 
204 			log.warn( "request() : Unkown named parameter '" + k + "'" );
205 		
206 		log.debug( conn.getRequestMethod() + " " + url );
207 		
208 		HttpResponse response = new HttpURLResponseAdapter(conn);
209 		if ( ContentType.ANY.equals( contentType ) ) contentType = conn.getContentType();
210 
211 		String method = conn.getRequestMethod();
212 		Object result = method.equals( "HEAD" ) || method.equals( "OPTIONS" ) ?
213 				null : parserRegistry.getAt( contentType ).call( response );
214 		
215 		log.debug( response.getStatusLine() );
216 		HttpResponseDecorator decoratedResponse = new HttpResponseDecorator( response, result );
217 		
218 		if ( log.isTraceEnabled() ) {
219 			for ( Header h : decoratedResponse.getHeaders() )
220 				log.trace( " << " + h.getName() + " : " + h.getValue() );
221 		}
222 				
223 		if ( conn.getResponseCode() > 399 ) 
224 			throw new HttpResponseException( decoratedResponse );
225 		
226 		return decoratedResponse;
227 	}

229 	private String getBasicAuthHeader( String user, String pass ) throws UnsupportedEncodingException {
230 	  return "Basic " + DefaultGroovyMethods.encodeBase64( 
231 			  (user + ":" + pass).getBytes("ISO-8859-1") ).toString();
232 	}

236 	 * Pass <code>null</code> to un-set authorization for this instance.
237 	 * @param user

242 		if ( user == null ) this.defaultHeaders.remove( "Authorization" );

286 		HttpURLConnection conn;

289 		HttpURLResponseAdapter( HttpURLConnection conn ) {

293 		public HttpEntity getEntity() {
294 			return new HttpEntity() {
295 
296 				public void consumeContent() throws IOException {
297 					conn.getInputStream().close();
298 				}
299 
300 				public InputStream getContent() 
301 						throws IOException, IllegalStateException {
302 					if ( Status.find( conn.getResponseCode() ) 
303 							== Status.FAILURE ) return conn.getErrorStream(); 
304 					return conn.getInputStream();
305 				}

307 				public Header getContentEncoding() {
308 					return new BasicHeader( "Content-Encoding",

310 				}

312 				public long getContentLength() {
313 					return conn.getContentLength();
314 				}
315 
316 				public Header getContentType() {
317 					return new BasicHeader( "Content-Type", conn.getContentType() );
318 				}
319 
320 				public boolean isChunked() {
321 					String enc = conn.getHeaderField( "Transfer-Encoding" );
322 					return enc != null && enc.contains( "chunked" );
323 				}
324 
325 				public boolean isRepeatable() {
326 					return false;
327 				}
328 
329 				public boolean isStreaming() {
330 					return true;
331 				}
332

334 					DefaultGroovyMethods.leftShift( out, conn.getInputStream() );
335 				}

354 		public boolean containsHeader( String key ) {

358 		public Header[] getAllHeaders() {
359 			if ( this.headers != null ) return this.headers;
360 			List<Header> headers = new ArrayList<Header>();
361 
362 			// see http://java.sun.com/j2se/1.5.0/docs/api/java/net/HttpURLConnection.html#getHeaderFieldKey(int)
363 			int i= conn.getHeaderFieldKey( 0 ) != null ? 0 : 1;
364 			String key;
365 			while ( ( key = conn.getHeaderFieldKey( i ) ) != null ) {
366 				headers.add( new BasicHeader( key, conn.getHeaderField( i++ ) ) );	
367 			}
368 
369 			this.headers = headers.toArray( new Header[headers.size()] ); 
370 			return this.headers;
371 		}
372 
373 		public Header getFirstHeader( String key ) {
374 			for ( Header h : getAllHeaders() )
375 				if ( h.getName().equals( key ) ) return h;
376 			return null;
377 		}

379 		/**
380 		 * Note that HttpURLConnection does not support multiple headers of 
381 		 * the same name.
382 		 */
383 		public Header[] getHeaders( String key ) {
384 			List<Header> headers = new ArrayList<Header>();
385 			for ( Header h : getAllHeaders() ) 
386 				if ( h.getName().equals( key ) ) headers.add( h );
387 			return headers.toArray( new Header[headers.size()] );
388 		}
