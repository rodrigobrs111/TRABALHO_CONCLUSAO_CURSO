60  /**
61   * <p>Keeps track of response parsers for each content type.  Each parser 
62   * should should be a closure that accepts an {@link HttpResponse} instance,
63   * and returns whatever handler is appropriate for reading the response 
64   * data for that content-type.  For example, a plain-text response should 
65   * probably be parsed with a <code>Reader</code>, while an XML response 
66   * might be parsed by an XmlSlurper, which would then be passed to the 
67   * response closure. </p>
68   * 
69   * <p>Note that all methods in this class assume {@link HttpResponse#getEntity()}
70   * return a non-null value.  It is the job of the HTTPBuilder instance to ensure
71   * a NullPointerException is not thrown by passing a response that contains no
72   * entity.</p>
73   * 
74   * @see ContentType
75   * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
76   */
77  public class ParserRegistry {
78

80  	 * The default parser used for unregistered content-types.  This is a copy 
81  	 * of {@link #parseStream(HttpResponse)}, which is like a no-op that just 
82  	 * returns the unaltered response stream.
83  	 */
84  	protected final Closure DEFAULT_PARSER = new MethodClosure( this, "parseStream" );

91  	/**
92  	 * This CatalogResolver is static to avoid the overhead of re-parsing
93  	 * the catalog definition file every time.  Unfortunately, there's no 
94  	 * way to share a single Catalog instance between resolvers.  The 
95  	 * {@link Catalog} class is technically not thread-safe, but as long as you 
96  	 * do not parse catalog files while using the resolver, it should be fine. 
97  	 */
98  	protected static CatalogResolver catalogResolver;
99  	
100 	static {
101 		CatalogManager catalogManager = new CatalogManager();
102 		catalogManager.setIgnoreMissingProperties( true );
103 		catalogManager.setUseStaticCatalog( false );
104 		catalogManager.setRelativeCatalogs( true );
105 		try {
106 			catalogResolver = new CatalogResolver( catalogManager );
107 			catalogResolver.getCatalog().parseCatalog( 
108 					ParserRegistry.class.getResource( "/catalog/html.xml" ) );
109 		} catch ( IOException ex ) {
110 			LogFactory.getLog( ParserRegistry.class )
111 				.warn( "Could not resolve default XML catalog", ex );
112 		}
113 	}

115 	/**
116 	 * Helper method to get the charset from the response.  This should be done 
117 	 * when manually parsing any text response to ensure it is decoded using the
118 	 * correct charset. For instance:<pre>
119 	 * Reader reader = new InputStreamReader( resp.getEntity().getContent(), 
120 	 *   ParserRegistry.getCharset( resp ) );</pre>
121 	 * @param resp
122 	 */
123 	public static String getCharset( HttpResponse resp ) {
124 		try {
125 			NameValuePair charset = resp.getEntity().getContentType()
126 				.getElements()[0].getParameterByName("charset");
127 			
128 			if ( charset == null || charset.getValue().trim().equals("") ) {
129 				log.warn( "Could not find charset in response" );
130 				return Charset.defaultCharset().name();
131 			}
132 				
133 			return charset.getValue();
134 		}
135 		catch ( RuntimeException ex ) { // NPE or OOB Exceptions
136 			log.warn( "Could not parse content-type header in response", ex );
137 			return Charset.defaultCharset().name();
138 		}
139 	}

141 	/**
142 	 * Helper method to get the content-type string from the response 
143 	 * (no charset).
144 	 * @param resp
145 	 */
146 	public static String getContentType( HttpResponse resp ) {
147 		if ( resp.getEntity() == null )
148 			throw new IllegalArgumentException( "Response does not contain data" );
149 		if ( resp.getEntity().getContentType() == null )
150 			throw new IllegalArgumentException( "Response does not have a content-type header" );
151 		try {
152 			return resp.getEntity().getContentType().getElements()[0].getName();
153 		}
154 		catch ( RuntimeException ex ) {  // NPE or OOB Exceptions
155 			throw new IllegalArgumentException( "Could not parse content-type from response", ex );
156 		}
157 	}

160 	 * Default parser used for binary data.  This simply returns the underlying
161 	 * response InputStream.
162 	 * @see ContentType#BINARY
163 	 * @see HttpEntity#getContent()
164 	 * @param resp
165 	 * @return an InputStream the binary response stream
166 	 * @throws IllegalStateException
167 	 * @throws IOException
168 	 */
169 	public InputStream parseStream( HttpResponse resp ) throws IOException {
170 		return resp.getEntity().getContent();
171 	}

173 	/**
174 	 * Default parser used to handle plain text data.  The response text 
175 	 * is decoded using the charset passed in the response content-type 
176 	 * header. 
177 	 * @see ContentType#TEXT
178 	 * @param resp
179 	 * @return
180 	 * @throws UnsupportedEncodingException
181 	 * @throws IllegalStateException
182 	 * @throws IOException
183 	 */
184 	public Reader parseText( HttpResponse resp ) throws IOException {
185 		return new InputStreamReader( resp.getEntity().getContent(), 
186 				ParserRegistry.getCharset( resp ) );
187 	}

189 	/**
190 	 * Default parser used to decode a URL-encoded response.
191 	 * @see ContentType#URLENC
192 	 * @param resp
193 	 * @return
194 	 * @throws IOException
195 	 */
196 	public Map<String,String> parseForm( HttpResponse resp ) throws IOException {
197 		List<NameValuePair> params = URLEncodedUtils.parse( resp.getEntity() );
198 		Map<String,String> paramMap = new HashMap<String,String>(params.size());
199 		for ( NameValuePair param : params ) 
200 			paramMap.put( param.getName(), param.getValue() );
201 		return paramMap;
202 	}

204 	/**
205 	 * Parse an HTML document by passing it through the NekoHTML parser.
206 	 * @see ContentType#HTML
207 	 * @see org.cyberneko.html.parsers.SAXParser
208 	 * @see XmlSlurper#parse(Reader)
209 	 * @param resp HTTP response from which to parse content
210 	 * @return the {@link GPathResult} from calling {@link XmlSlurper#parse(Reader)}
211 	 * @throws IOException
212 	 * @throws SAXException
213 	 */
214 	public GPathResult parseHTML( HttpResponse resp ) throws IOException, SAXException {
215 		XMLReader p = new org.cyberneko.html.parsers.SAXParser();
216 		p.setEntityResolver( catalogResolver );
217 		return new XmlSlurper( p ).parse( parseText( resp ) );
218 	}

221 	 * Default parser used to decode an XML response.  
222 	 * @see ContentType#XML
223 	 * @see XmlSlurper#parse(Reader)
224 	 * @param resp HTTP response from which to parse content
225 	 * @return the {@link GPathResult} from calling {@link XmlSlurper#parse(Reader)}
226 	 * @throws IOException
227 	 * @throws SAXException
228 	 * @throws ParserConfigurationException
229 	 */
230 	public GPathResult parseXML( HttpResponse resp ) throws IOException, SAXException, ParserConfigurationException {
231 		XmlSlurper xml = new XmlSlurper();
232 		xml.setEntityResolver( catalogResolver );
233 		return xml.parse( parseText( resp ) );
234 	}

249 	/**
250 	 * Returns a map of default parsers.  Override this method to change 
251 	 * what parsers are registered by default.  You can of course call
252 	 * <code>super.buildDefaultParserMap()</code> and then add or remove 
253 	 * from that result as well.
254 	 */
255 	protected Map<String,Closure> buildDefaultParserMap() {
256 		Map<String,Closure> parsers = new HashMap<String,Closure>();
257 		
258 		parsers.put( ContentType.BINARY.toString(), new MethodClosure( this, "parseStream" ) );
259 		parsers.put( ContentType.TEXT.toString(), new MethodClosure(this,"parseText") );
260 		parsers.put( ContentType.URLENC.toString(), new MethodClosure(this,"parseForm") );
261 		parsers.put( ContentType.HTML.toString(), new MethodClosure(this,"parseHTML") );
262 		
263 		Closure pClosure = new MethodClosure(this,"parseXML");
264 		for ( String ct : ContentType.XML.getContentTypeStrings() )
265 			parsers.put( ct, pClosure );
266 		
267 		pClosure = new MethodClosure(this,"parseJSON");
268 		for ( String ct : ContentType.JSON.getContentTypeStrings() )
269 			parsers.put( ct, pClosure );
270 		
271 		return parsers;
272 	}

311 	/** 
312 	 * Retrieve a parser for the given response content-type string.  This
313 	 * is called by HTTPBuildre to retrieve the correct parser for a given 
314 	 * content-type.  The parser is then used to decode the response data prior
315 	 * to passing it to a response handler. 
316 	 * @param contentType
317 	 * @return parser that can interpret the given response content type,
318 	 *   or the default parser if no parser is registered for the given 
319 	 *   content-type.  It should NOT return a null value.
320 	 */
321 	public Closure getAt( Object contentType ) {
322 		String ct = contentType.toString();
323 		int idx = ct.indexOf( ';' ); 
324 		if ( idx > 0 ) ct = ct.substring( 0, idx );
325 		
326 		Closure parser = registeredParsers.get(ct);
327 		if ( parser != null ) return parser;
328 
329 		log.warn( "Cannot find parser for content-type: " + ct 
330 					+ " -- using default parser.");
331 		return defaultParser;
332 	}

334 	/**
335 	 * Register a new parser for the given content-type.  The parser closure
336 	 * should accept an {@link HttpResponse} argument and return a type suitable
337 	 * to be passed as the 'parsed data' argument of a 
338 	 * {@link RequestConfigDelegate#getResponse() response handler} closure.
339 	 * @param contentType  <code>content-type</code> string
340 	 * @param value code that will parse the HttpResponse and return parsed 
341 	 *   data to the response handler. 
342 	 */
343 	public void putAt( Object contentType, Closure value ) {
344 		if ( contentType instanceof ContentType ) {
345 			for ( String ct : ((ContentType)contentType).getContentTypeStrings() )
346 				this.registeredParsers.put( ct, value );
347 		}
348 		else this.registeredParsers.put( contentType.toString(), value );
349 	}
