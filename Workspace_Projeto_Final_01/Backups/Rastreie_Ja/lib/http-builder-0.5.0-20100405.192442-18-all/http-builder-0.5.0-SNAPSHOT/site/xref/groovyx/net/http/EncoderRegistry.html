63  /**
64   * <p>This class handles creation of the request body (i.e. for a 
65   * PUT or POST operation) based on content-type.   When a 
66   * {@link RequestConfigDelegate#setBody(Object) body} is set from the builder, it is 
67   * processed based on the {@link RequestConfigDelegate#getRequestContentType() 
68   * request content-type}.  For instance, the {@link #encodeForm(Map)} method 
69   * will be invoked if the request content-type is form-urlencoded, which will 
70   * cause the following:<code>body=[a:1, b:'two']</code> to be encoded as 
71   * the equivalent <code>a=1&b=two</code> in the request body.</p>
72   * 
73   * <p>Most default encoders can handle a closure as a request body.  In this 
74   * case, the closure is executed and a suitable 'builder' passed to the 
75   * closure that is  used for constructing the content.  In the case of 
76   * binary encoding this would be an OutputStream; for TEXT encoding it would
77   * be a PrintWriter, and for XML it would be an already-bound 
78   * {@link StreamingMarkupBuilder}. See each <code>encode...</code> method 
79   * for details for each particular content-type.</p>
80   * 
81   * <p>Contrary to its name, this class does not have anything to do with the 
82   * <code>content-encoding</code> HTTP header.  </p>
83   * 
84   * @see RequestConfigDelegate#setBody(Object)
85   * @see RequestConfigDelegate#send(Object, Object)
86   * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
87   */

103 	/**
104 	 * Default request encoder for a binary stream.  Acceptable argument 
105 	 * types are:
106 	 * <ul>
107 	 *   <li>InputStream</li>
108 	 *   <li>byte[] / ByteArrayOutputStream</li>
109 	 *   <li>Closure</li>
110 	 * </ul>
111 	 * If a closure is given, it is executed with an OutputStream passed
112 	 * as the single closure argument.  Any data sent to the stream from the 
113 	 * body of the closure is used as the request content body.
114 	 * @param data
115 	 * @return an {@link HttpEntity} encapsulating this request data
116 	 * @throws UnsupportedEncodingException
117 	 */
118 	public InputStreamEntity encodeStream( Object data ) throws UnsupportedEncodingException {
119 		InputStreamEntity entity = null;
120 		
121 		if ( data instanceof ByteArrayInputStream ) {
122 			// special case for ByteArrayIS so that we can set the content length.
123 			ByteArrayInputStream in = ((ByteArrayInputStream)data);
124 			entity = new InputStreamEntity( in, in.available() );
125 		}
126 		else if ( data instanceof InputStream ) {
127 			entity = new InputStreamEntity( (InputStream)data, -1 );
128 		}
129 		else if ( data instanceof byte[] ) {
130 			byte[] out = ((byte[])data); 
131 			entity = new InputStreamEntity( new ByteArrayInputStream(
132 					out), out.length );
133 		}
134 		else if ( data instanceof ByteArrayOutputStream ) {
135 			ByteArrayOutputStream out = ((ByteArrayOutputStream)data); 
136 			entity = new InputStreamEntity( new ByteArrayInputStream(
137 					out.toByteArray()), out.size() );
138 		}
139 		else if ( data instanceof Closure ) {
140 			ByteArrayOutputStream out = new ByteArrayOutputStream();
141 			((Closure)data).call( out ); // data is written to out
142 			entity = new InputStreamEntity( new ByteArrayInputStream( 
143 					out.toByteArray()), out.size() );
144 		}
145 
146 		if ( entity == null ) throw new IllegalArgumentException( 
147 				"Don't know how to encode " + data + " as a byte stream" );
148 		
149 		entity.setContentType( ContentType.BINARY.toString() );
150 		return entity;
151 	}

153 	/**
154 	 * Default handler used for a plain text content-type.  Acceptable argument
155 	 * types are:
156 	 * <ul>
157 	 *   <li>Closure</li>
158 	 *   <li>Writable</li>
159 	 *   <li>Reader</li>
160 	 * </ul>
161 	 * For Closure argument, a {@link PrintWriter} is passed as the single 
162 	 * argument to the closure.  Any data sent to the writer from the 
163 	 * closure will be sent to the request content body.
164 	 * @param data
165 	 * @return an {@link HttpEntity} encapsulating this request data
166 	 * @throws IOException
167 	 */
168 	public HttpEntity encodeText( Object data ) throws IOException {
169 		if ( data instanceof Closure ) {
170 			StringWriter out = new StringWriter();
171 			PrintWriter writer = new PrintWriter( out );
172 			((Closure)data).call( writer );
173 			writer.close();
174 			out.flush();
175 			data = out;
176 		}
177 		else if ( data instanceof Writable ) {
178 			StringWriter out = new StringWriter();
179 			((Writable)data).writeTo(out);
180 			out.flush();
181 			data = out;
182 		}
183 		else if ( data instanceof Reader && ! (data instanceof BufferedReader) )
184 			data = new BufferedReader( (Reader)data );
185 		if ( data instanceof BufferedReader ) {
186 			StringWriter out = new StringWriter();
187 			DefaultGroovyMethods.leftShift( out, (BufferedReader)data );
188 			
189 			data = out;
190 		}
191 		// if data is a String, we are already covered.
192 		return createEntity( ContentType.TEXT, data.toString() );
193 	}

195 	/**
196 	 * Set the request body as a url-encoded list of parameters.  This is 
197 	 * typically used to simulate a HTTP form POST. 
198 	 * For multi-valued parameters, enclose the values in a list, e.g. 
199 	 * <pre>[ key1 : ['val1', 'val2'], key2 : 'etc.' ]</pre>
200 	 * @param params
201 	 * @return an {@link HttpEntity} encapsulating this request data
202 	 * @throws UnsupportedEncodingException
203 	 */
204 	public UrlEncodedFormEntity encodeForm( Map<?,?> params ) 
205 			throws UnsupportedEncodingException {
206 		List<NameValuePair> paramList = new ArrayList<NameValuePair>();
207 
208 		for ( Object key : params.keySet() ) {
209 			Object val = params.get( key );
210 			if ( val instanceof List ) 
211 				for ( Object subVal : (List)val ) 
212 					paramList.add( new BasicNameValuePair( key.toString(), 
213 							( subVal == null ) ? "" : subVal.toString() ) );
214 
215 			else paramList.add( new BasicNameValuePair( key.toString(), 
216 					( val == null ) ? "" : val.toString() ) );
217 		}
218 			
219 		return new UrlEncodedFormEntity( paramList, charset.name() );
220 	}

236 	 * Encode the content as XML.  The argument may be either an object whose 
237 	 * <code>toString</code> produces valid markup, or a Closure which will be 
238 	 * interpreted as a builder definition.  A closure argument is 
239 	 * passed to {@link StreamingMarkupBuilder#bind(groovy.lang.Closure)}.
240 	 * @param xml data that defines the XML structure
241 	 * @return an {@link HttpEntity} encapsulating this request data
242 	 * @throws UnsupportedEncodingException
243 	 */
244 	public HttpEntity encodeXML( Object xml ) throws UnsupportedEncodingException {
245 		if ( xml instanceof Closure ) {
246 			StreamingMarkupBuilder smb = new StreamingMarkupBuilder();
247 			xml = smb.bind( xml );
248 		}
249 		return createEntity( ContentType.XML, xml.toString() );
250 	}

252 	/**
253 	 * <p>Accepts a Collection or a JavaBean object which is converted to JSON.  
254 	 * A Map or POJO/POGO will be converted to a {@link JSONObject}, and any 
255 	 * other collection type will be converted to a {@link JSONArray}.  A
256 	 * String or GString will be interpreted as valid JSON and passed directly
257 	 * as the request body (with charset conversion if necessary.)</p> 
258 	 * 
259 	 * <p>If a Closure is passed as the model, it will be executed as if it were 
260 	 * a JSON object definition passed to a {@link JsonGroovyBuilder}.  In order
261 	 * for the closure to be interpreted correctly, there must be a 'root' 
262 	 * element immediately inside the closure.  For example:</p>
263 	 * 
264 	 * <pre>builder.post( JSON ) {
265 	 *   body = {
266 	 *     root {
267 	 *       first {
268 	 *         one = 1
269 	 *         two = '2'
270 	 *       }
271 	 *       second = 'some string'
272 	 *     }
273 	 *   }
274 	 * }</pre>
275 	 * <p> will return the following JSON string:<pre>
276 	 * {"root":{"first":{"one":1,"two":"2"},"second":"some string"}}</pre></p>
277 	 *  
278 	 * @param model data to be converted to JSON, as specified above.
279 	 * @return an {@link HttpEntity} encapsulating this request data
280 	 * @throws UnsupportedEncodingException
281 	 */
282 	@SuppressWarnings("unchecked")
283 	public HttpEntity encodeJSON( Object model ) throws UnsupportedEncodingException {
284 		
285 		Object json;	
286 		if ( model instanceof Map ) {
287 			json = new JSONObject();
288 			((JSONObject)json).putAll( (Map)model );
289 		}
290 		else if ( model instanceof Collection ) {
291 			json = new JSONArray();
292 			((JSONArray)json).addAll( (Collection)model );
293 		}
294 		else if ( model instanceof Closure ) {
295 			Closure closure = (Closure)model;
296 			closure.setDelegate( new JsonGroovyBuilder() );
297 			json = (JSON)closure.call();
298 		}
299 		else if ( model instanceof String || model instanceof GString )
300 			json = model; // assume string is valid JSON already.
301 		else json = JSONObject.fromObject( model ); // Assume object is a JavaBean
302 		
303 		return this.createEntity( ContentType.JSON, json.toString() );
304 	}

307 	 * Helper method used by encoder methods to create an {@link HttpEntity} 
308 	 * instance that encapsulates the request data.  This may be used by any 
309 	 * non-streaming encoder that needs to send textual data.  It also sets the 
310 	 * {@link #setCharset(String) charset} portion of the content-type header. 
311 	 * 
312 	 * @param ct content-type of the data
313 	 * @param data textual request data to be encoded 
314 	 * @return an instance to be used for the 
315 	 *  {@link HttpEntityEnclosingRequest#setEntity(HttpEntity) request content} 
316 	 * @throws UnsupportedEncodingException
317 	 */
318 	protected StringEntity createEntity( ContentType ct, String data ) 
319 			throws UnsupportedEncodingException {
320 		StringEntity entity = new StringEntity( data, charset.toString() );
321 		entity.setContentType( ct.toString() );
322 		return entity;
323 	}

325 	/**
326 	 * Returns a map of default encoders.  Override this method to change 
327 	 * what encoders are registered by default.  You can of course call
328 	 * <code>super.buildDefaultEncoderMap()</code> and then add or remove 
329 	 * from that result as well.
330 	 */
331 	protected Map<String,Closure> buildDefaultEncoderMap() {
332 		Map<String,Closure> encoders = new HashMap<String,Closure>();
333 		
334 		encoders.put( ContentType.BINARY.toString(), new MethodClosure(this,"encodeStream") );
335 		encoders.put( ContentType.TEXT.toString(), new MethodClosure( this, "encodeText" ) );
336 		encoders.put( ContentType.URLENC.toString(), new MethodClosure( this, "encodeForm" ) );
337 		
338 		Closure encClosure = new MethodClosure(this,"encodeXML");
339 		for ( String ct : ContentType.XML.getContentTypeStrings() )
340 			encoders.put( ct, encClosure );
341 		encoders.put( ContentType.HTML.toString(), encClosure );
342 		
343 		encClosure = new MethodClosure(this,"encodeJSON");
344 		for ( String ct : ContentType.JSON.getContentTypeStrings() )
345 			encoders.put( ct, encClosure );
346 		
347 		return encoders;
348 	}

350 	/** 
351 	 * Retrieve a encoder for the given content-type.  This
352 	 * is called by HTTPBuilder to retrieve the correct encoder for a given 
353 	 * content-type.  The encoder is then used to serialize the request data 
354 	 * in the request body. 
355 	 * @param contentType
356 	 * @return encoder that can interpret the given content type,
357 	 *   or null.
358 	 */
359 	public Closure getAt( Object contentType ) {
360 		String ct = contentType.toString();
361 		int idx = ct.indexOf( ';' ); 
362 		if ( idx > 0 ) ct = ct.substring( 0, idx );
363 		
364 		return registeredEncoders.get(ct);
365 	}

367 	/** 
368 	 * Register a new encoder for the given content type.  If any encoder 
369 	 * previously existed for that content type it will be replaced.  The 
370 	 * closure must return an {@link HttpEntity}.  It will also usually 
371 	 * accept a single argument, which will be whatever is set in the request
372 	 * configuration closure via {@link RequestConfigDelegate#setBody(Object)}.
373 	 * @param contentType
374 	 * @param closure
375 	 */
376 	public void putAt( Object contentType, Closure value ) {
377 		if ( contentType instanceof ContentType ) {
378 			for ( String ct : ((ContentType)contentType).getContentTypeStrings() )
379 				this.registeredEncoders.put( ct, value );
380 		}
381 		else this.registeredEncoders.put( contentType.toString(), value );
382 	}
