37  /**
38   * This class implements a mutable URI.  All <code>set</code>, <code>add</code> 
39   * and <code>remove</code> methods affect this class' internal URI 
40   * representation.  All mutator methods support chaining, e.g.
41   * <pre>
42   * new URIBuilder("http://www.google.com/")
43   *   .setScheme( "https" )
44   *   .setPort( 443 )
45   *   .setPath( "some/path" )
46   *   .toString();
47   * </pre>
48   * A slightly more 'Groovy' version would be:
49   * <pre>
50   * new URIBuilder('http://www.google.com/').with {
51   *    scheme = 'https'
52   *    port = 443
53   *    path = 'some/path'
54   *    query = [p1:1, p2:'two']
55   * }.toString()
56   * </pre>
57   * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
58   */
59  public class URIBuilder implements Cloneable {

82  	 * Utility method to convert a number of type to a URI instance. 
83  	 * @param uri a {@link URI}, {@link URL} or any object that produces a 
84  	 *   valid URI string from its <code>toString()</code> result.
85  	 * @return a valid URI parsed from the given object
86  	 * @throws URISyntaxException
87  	 */
88  	public static URI convertToURI( Object uri ) throws URISyntaxException {
89  		if ( uri instanceof URI ) return (URI)uri;
90  		if ( uri instanceof URL ) return ((URL)uri).toURI();
91  		if ( uri instanceof URIBuilder ) return ((URIBuilder)uri).toURI();
92  		return new URI( uri.toString() ); // assume any other object type produces a valid URI string
93  	}

124 	 * relative to the current path.
125 	 * e.g. <pre>
126 	 *   def uri = new URIBuilder( 'http://localhost/p1/p2?a=1' )
127 	 *

133 	 *

135 	 *   assert uri.toString() == 'http://localhost/p4?a=1&b=2&c=3#frag'
136 	 * <pre>
137 	 * @param path the path portion of this URI, relative to the current URI.
138 	 * @return this URIBuilder instance, for method chaining.
139 	 * @throws URISyntaxException if the given path contains characters that 
140 	 *   cannot be converted to a valid URI

143 		this.base = base.resolve( new URI(null,null, path, base.getQuery(), base.getFragment()) );
144 //		path = base.resolve( path ).getPath();
145 //		this.base = new URI( base.getScheme(), base.getUserInfo(),

151 	/* TODO null/ zero-size check if this is ever made public */
152 	protected URIBuilder setQueryNVP( List<NameValuePair> nvp ) throws URISyntaxException {
153 		/* Passing the query string in the URI constructor will 
154 		 * double-escape query parameters and goober things up.  So we have 
155 		 * to create a full path+query+fragment and use URI#resolve() to 
156 		 * create the new URI.  */
157 		StringBuilder sb = new StringBuilder();
158 		String path = base.getRawPath();
159 		if ( path != null ) sb.append( path );
160 		sb.append( '?' );
161 		sb.append( URLEncodedUtils.format( nvp, ENC ) ); 
162 		String frag = base.getRawFragment();
163 		if ( frag != null ) sb.append( '#' ).append( frag );
164 		this.base = base.resolve( sb.toString() );
165 
166 		return this;
167 	}

169 	/**
170 	 * Set the query portion of the URI.  For query parameters with multiple 
171 	 * values, put the values in a list like so:
172 	 * <pre>uri.query = [ p1:'val1', p2:['val2', 'val3'] ]
173 	 * // will produce a query string of ?p1=val1&p2=val2&p2=val3</pre>
174 	 * 
175 	 * @param params a Map of parameters that will be transformed into the query string
176 	 * @return this URIBuilder instance, for method chaining.
177 	 * @throws URISyntaxException
178 	 */
179 	public URIBuilder setQuery( Map<?,?> params ) throws URISyntaxException {
180 		if ( params == null || params.size() < 1 ) {
181 			this.base = new URI( base.getScheme(), base.getUserInfo(), 
182 				base.getHost(), base.getPort(), base.getPath(),
183 				null, base.getFragment() );
184 		}
185 		else {
186 			List<NameValuePair> nvp = new ArrayList<NameValuePair>(params.size());
187 			for ( Object key : params.keySet() ) {
188 				Object value = params.get(key);
189 				if ( value instanceof List ) {
190 					for (Object val : (List)value )
191 						nvp.add( new BasicNameValuePair( key.toString(), 
192 								( val != null ) ? val.toString() : "" ) );
193 				}
194 				else nvp.add( new BasicNameValuePair( key.toString(), 
195 						( value != null ) ? value.toString() : "" ) );
196 			}
197 			this.setQueryNVP( nvp );
198 		}
199 		return this;
200 	}

202 	/**
203 	 * Get the query string as a map for convenience.  If any parameter contains
204 	 * multiple values (e.g. <code>p1=one&p1=two</code>) both values will be 
205 	 * inserted into a list for that paramter key (<code>[p1 : ['one','two']]
206 	 * </code>).  Note that this is not a "live" map.  Therefore, you cannot 
207 	 * call 
208 	 * <pre> uri.query.a = 'BCD'</pre>
209 	 * You will not modify the query string but instead the generated map of
210 	 * parameters.  Instead, you need to use {@link #removeQueryParam(String)}
211 	 * first, then {@link #addQueryParam(String, Object)}, or call 
212 	 * {@link #setQuery(Map)} which will set the entire query string.
213 	 * @return a map of String name/value pairs representing the URI's query 
214 	 * string.
215 	 */
216 	public Map<String,Object> getQuery() {
217 		Map<String,Object> params = new HashMap<String,Object>();		
218 		List<NameValuePair> pairs = this.getQueryNVP();
219 		
220 		for ( NameValuePair pair : pairs ) {
221 			
222 			String key = pair.getName();
223 			Object existing = params.get( key );
224 
225 			if ( existing == null ) params.put( key, pair.getValue() );
226 
227 			else if ( existing instanceof List ) 
228 				((List)existing).add( pair.getValue() );
229 
230 			else {
231 				List<String> vals = new ArrayList<String>(2);
232 				vals.add( (String)existing );
233 				vals.add( pair.getValue() );
234 				params.put( key, vals );
235 			}
236 		}
237 		
238 		return params;
239 	}

241 	protected List<NameValuePair> getQueryNVP() {
242 		List<NameValuePair> nvps = URLEncodedUtils.parse( this.base, ENC );
243 		List<NameValuePair> newList = new ArrayList<NameValuePair>();
244 		if ( nvps != null ) newList.addAll( nvps );
245 		return newList;
246 	}

260 	 * Remove the given query parameter from this URI's query string.
261 	 * @param param the query name to remove 
262 	 * @return this URIBuilder instance, for method chaining.
263 	 * @throws URISyntaxException
264 	 */
265 	public URIBuilder removeQueryParam( String param ) throws URISyntaxException {
266 		List<NameValuePair> params = getQueryNVP();
267 		NameValuePair found = null;
268 		for ( NameValuePair nvp : params )  // BOO linear search.  Assume the list is small.
269 			if ( nvp.getName().equals( param ) ) {
270 				found = nvp;
271 				break;
272 			}
273 		
274 		if ( found == null ) throw new IllegalArgumentException( "Param '" + param + "' not found" );
275 		params.remove( found );
276 		this.setQueryNVP( params );
277 		return this;
278 	}

288 	 * This will append a query parameter to the existing query string.  If the given 
289 	 * parameter is already part of the query string, it will be appended to.  
290 	 * To replace the existing value of a certain parameter, either call 
291 	 * {@link #removeQueryParam(String)} first, or use {@link #getQuery()},
292 	 * modify the value in the map, then call {@link #setQuery(Map)}.
293 	 * @param param query parameter name 
294 	 * @param value query parameter value (will be converted to a string if 
295 	 *   not null.  If <code>value</code> is null, it will be set as the empty 
296 	 *   string.
297 	 * @return this URIBuilder instance, for method chaining.
298 	 * @throws URISyntaxException if the query parameter values cannot be 
299 	 * converted to a valid URI.
300 	 * @see #setQuery(Map) 
301 	 */
302 	public URIBuilder addQueryParam( String param, Object value ) throws URISyntaxException {
303 		List<NameValuePair> params = getQueryNVP();
304 		params.add( new BasicNameValuePair( param, 
305 				( value != null ) ? value.toString() : "" ) );
306 		this.setQueryNVP( params );
307 		return this;

318 	 * Add these parameters to the URIBuilder's existing query string.
319 	 * Parameters may be passed either as a single map argument, or as a list
320 	 * of named arguments.  e.g. 
321 	 * <pre> uriBuilder.addQueryParams( [one:1,two:2] )
322 	 * uriBuilder.addQueryParams( three : 3 ) </pre>
323 	 * 
324 	 * If any of the parameters already exist in the URI query, these values 
325 	 * will <strong>not</strong> replace them.  Multiple values for the same 
326 	 * query parameter may be added by putting them in a list. See 
327 	 * {@link #setQuery(Map)}.
328 	 * 
329 	 * @param params parameters to add to the existing URI query (if any).
330 	 * @return this URIBuilder instance, for method chaining.
331 	 * @throws URISyntaxException
332 	 */
333 	@SuppressWarnings("unchecked")
334 	public URIBuilder addQueryParams( Map<?,?> params ) throws URISyntaxException {
335 		List<NameValuePair> nvp = new ArrayList<NameValuePair>();
336 		for ( Object key : params.keySet() ) {
337 			Object value = params.get( key );
338 			if ( value instanceof List ) {
339 				for ( Object val : (List)value )
340 					nvp.add( new BasicNameValuePair( key.toString(), 
341 							( val != null ) ? val.toString() : "" ) );
342 			}
343 			else nvp.add( new BasicNameValuePair( key.toString(), 
344 					( value != null ) ? value.toString() : "" ) );
345 		}
346 		this.addQueryParams( nvp );
347 		return this;
348 	}
