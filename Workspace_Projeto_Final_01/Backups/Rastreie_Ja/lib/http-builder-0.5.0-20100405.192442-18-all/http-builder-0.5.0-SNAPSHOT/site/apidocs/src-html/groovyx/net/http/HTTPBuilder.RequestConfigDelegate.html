062    /** <p>
063     * Groovy DSL for easily making HTTP requests, and handling request and response
064     * data.  This class adds a number of convenience mechanisms built on top of 
065     * Apache HTTPClient for things like URL-encoded POSTs and REST requests that 
066     * require building and parsing JSON or XML.  Convenient access to a few common
067     * authentication methods is also available.</p>
068     * 
069     * 
070     * <h3>Conventions</h3>
071     * <p>HTTPBuilder has properties for default headers, URI, contentType, etc.  
072     * All of these values are also assignable (and in many cases, in much finer 
073     * detail) from the {@link RequestConfigDelegate} as well.  In any cases where the value
074     * is not set on the delegate (from within a request closure,) the builder's 
075     * default value is used.  </p>
076     * 
077     * <p>For instance, any methods that do not take a <code>uri</code> parameter 
078     * assume you will set the <code>uri</code> property in the request closure or 
079     * use HTTPBuilder's assigned {@link #getUri() default URI}.</p>
080     * 
081     * 
082     * <h3>Response Parsing</h3>
083     * <p>By default, HTTPBuilder uses {@link ContentType#ANY} as the default 
084     * content-type.  This means the value of the request's <code>Accept</code> 
085     * header is <code>&#42;/*</code>, and the response parser is determined 
086     * based on the response <code>content-type</code> header. </p>
087     * 
088     * <p><strong>If</strong> any contentType is given (either in 
089     * {@link #setContentType(Object)} or as a request method parameter), the 
090     * builder will attempt to parse the response using that content-type, 
091     * regardless of what the server actually responds with.  </p>
092     * 
093     *  
094     * <h3>Examples:</h3>
095     * Perform an HTTP GET and print the response:
096     * <pre>
097     *   def http = new HTTPBuilder('http://www.google.com')
098     *   
099     *   http.get( path : '/search', 
100     *             contentType : TEXT,
101     *             query : [q:'Groovy'] ) { resp, reader ->
102     *     println "response status: ${resp.statusLine}"
103     *     println 'Response data: -----'
104     *     System.out << reader
105     *     println '\n--------------------'
106     *   }
107     * </pre>
108     *   
109     * Long form for other HTTP methods, and response-code-specific handlers.  
110     * This is roughly equivalent to the above example.
111     *   
112     * <pre>
113     *   def http = new HTTPBuilder('http://www.google.com/search?q=groovy')
114     *   
115     *   http.request( GET, TEXT ) { req ->
116     *   
117     *     // executed for all successful responses:
118     *     response.success = { resp, reader ->
119     *       println 'my response handler!'
120     *       assert resp.statusLine.statusCode == 200
121     *       println resp.statusLine
122     *       System.out << reader // print response stream
123     *     }
124     *     
125     *     // executed only if the response status code is 401:
126     *     response.'404' = { resp -> 
127     *       println 'not found!'
128     *     }
129     *   }
130     * </pre>
131     *   
132     * You can also set a default response handler called for any status
133     * code > 399 that is not matched to a specific handler. Setting the value
134     * outside a request closure means it will apply to all future requests with
135     * this HTTPBuilder instance:
136     * <pre>
137     *   http.handler.failure = { resp ->
138     *     println "Unexpected failure: ${resp.statusLine}"
139     *   }
140     * </pre>
141     *   
142     *   
143     * And...  Automatic response parsing for registered content types!
144     *   
145     * <pre>
146     *   http.request( 'http://ajax.googleapis.com', GET, JSON ) {
147     *     uri.path = '/ajax/services/search/web'
148     *     uri.query = [ v:'1.0', q: 'Calvin and Hobbes' ]
149     *     
150     *     response.success = { resp, json ->
151     *       assert json.size() == 3
152     *       println "Query response: "
153     *       json.responseData.results.each {
154     *         println "  ${it.titleNoFormatting} : ${it.visibleUrl}"
155     *       }
156     *     }
157     *   }
158     * </pre>
159     * 
160     * 
161     * @author <a href='mailto:tomstrummer+httpbuilder@gmail.com'>Tom Nichols</a>
162     */

404            /**
405             * Create a {@link RequestConfigDelegate} from the given arguments, execute the 
406             * config closure, then pass the delegate to {@link #doRequest(RequestConfigDelegate)},
407             * which actually executes the request.
408             */
409            protected Object doRequest( URI uri, Method method, Object contentType, Closure configClosure ) 
410                            throws ClientProtocolException, IOException {
411    
412                    HttpRequestBase reqMethod;
413                    try { reqMethod = method.getRequestType().newInstance();
414                    // this exception should reasonably never occur:
415                    } catch ( Exception e ) { throw new RuntimeException( e ); }
416    
417                    reqMethod.setURI( uri );
418                    RequestConfigDelegate delegate = new RequestConfigDelegate( reqMethod, contentType, 
419                                    this.defaultRequestHeaders,
420                                    this.defaultResponseHandlers );         
421                    configClosure.setDelegate( delegate );
422                    configClosure.setResolveStrategy( Closure.DELEGATE_FIRST );
423                    configClosure.call( reqMethod );
424                    
425                    return this.doRequest( delegate );
426            }

428            /**
429             * All <code>request</code> methods delegate to this method.
430             */
431            protected Object doRequest( final RequestConfigDelegate delegate ) 
432                            throws ClientProtocolException, IOException {
433                    
434                    final HttpRequestBase reqMethod = delegate.getRequest();
435                    
436                    Object contentType = delegate.getContentType();
437                    String acceptContentTypes = contentType.toString();
438                    if ( contentType instanceof ContentType ) 
439                            acceptContentTypes = ((ContentType)contentType).getAcceptHeader();
440                    
441                    reqMethod.setHeader( "Accept", acceptContentTypes );
442                    reqMethod.setURI( delegate.getUri().toURI() );
443                    log.debug( reqMethod.getMethod() + " " + reqMethod.getURI() );
444    
445                    // set any request headers from the delegate
446                    Map<?,?> headers = delegate.getHeaders(); 
447                    for ( Object key : headers.keySet() ) {
448                            Object val = headers.get( key );
449                            if ( key == null ) continue;
450                            if ( val == null ) reqMethod.removeHeaders( key.toString() ); 
451                            else reqMethod.setHeader( key.toString(), val.toString() );
452                    }
453                    
454                    HttpResponseDecorator resp = new HttpResponseDecorator( 
455                                    client.execute( reqMethod ), null);
456                    try {
457                            int status = resp.getStatusLine().getStatusCode();
458                            Closure responseClosure = delegate.findResponseHandler( status );
459                            log.debug( "Response code: " + status + "; found handler: " + responseClosure );
460                            
461                            Object[] closureArgs = null;
462                            switch ( responseClosure.getMaximumNumberOfParameters() ) {
463                            case 1 :
464                                    closureArgs = new Object[] { resp };
465                                    break;
466                            case 2 : // parse the response entity if the response handler expects it:
467                                    HttpEntity entity = resp.getEntity();
468                                    try {
469                                            if ( entity == null || entity.getContentLength() == 0 ) 
470                                                    closureArgs = new Object[] { resp, null };
471                                            else closureArgs = new Object[] { resp, parseResponse( resp, contentType ) };
472                                    }
473                                    catch ( Exception ex ) {
474                                            Header h = entity.getContentType();
475                                            String respContentType = h != null ? h.getValue() : null;
476                                            log.warn( "Error parsing '" + respContentType + "' response", ex );
477                                            throw new ResponseParseException( resp, ex );   
478                                    }
479                                    break;
480                            default:
481                                    throw new IllegalArgumentException( 
482                                                    "Response closure must accept one or two parameters" );
483                            }
484                            
485                            Object returnVal = responseClosure.call( closureArgs );
486                            log.trace( "response handler result: " + returnVal );
487                            
488                            return returnVal;
489                    }
490                    finally {
491                            HttpEntity entity = resp.getEntity(); 
492                            if ( entity != null ) entity.consumeContent();
493                    }
494            }

496            /**
497             * Parse the response data based on the given content-type.  
498             * If the given content-type is {@link ContentType#ANY}, the 
499             * <code>content-type</code> header from the response will be used to 
500             * determine how to parse the response. 
501             * @param resp
502             * @param contentType
503             * @return whatever was returned from the parser retrieved for the given 
504             *  content-type, or <code>null</code> if no parser could be found for this 
505             *  content-type.  The parser will also return <code>null</code> if the 
506             *  response does not contain any content (e.g. in response to a HEAD request).
507             * @throws HttpResponseException if there is a error parsing the response
508             */
509            protected Object parseResponse( HttpResponse resp, Object contentType ) 
510                            throws HttpResponseException {
511                    // For HEAD or OPTIONS requests, there should be no response entity.
512                    if ( resp.getEntity() == null ) {
513                            log.debug( "Response contains no entity.  Parsed data is null." );
514                            return null;
515                    }
516                    // first, start with the _given_ content-type
517                    String responseContentType = contentType.toString();
518                    // if the given content-type is ANY ("*/*") then use the response content-type
519                    try {
520                            if ( ContentType.ANY.toString().equals( responseContentType ) )
521                                    responseContentType = ParserRegistry.getContentType( resp );
522                    }
523                    catch ( RuntimeException ex ) {
524                            log.warn( "Could not parse content-type: " + ex.getMessage() );
525                            /* if for whatever reason we can't determine the content-type, but
526                             * still want to attempt to parse the data, use the BINARY 
527                             * content-type so that the response will be buffered into a 
528                             * ByteArrayInputStream. */
529                            responseContentType = ContentType.BINARY.toString();
530                    }
531                    
532                    Object parsedData = null;
533                    Closure parser = parsers.getAt( responseContentType );
534                    if ( parser == null ) log.warn( "No parser found for content-type: " 
535                            + responseContentType );
536                    else {
537                            log.debug( "Parsing response as: " + responseContentType );
538                            parsedData = parser.call( resp );
539                            if ( parsedData == null ) log.warn( "Parser returned null!" );
540                            else log.debug( "Parsed data to instance of: " + parsedData.getClass() );
541                    }
542                    return parsedData;
543            }

546             * Creates default response handlers for {@link Status#SUCCESS success} and
547             * {@link Status#FAILURE failure} status codes.  This is used to populate 
548             * the handler map when a new HTTPBuilder instance is created. 
549             * @see #defaultSuccessHandler(HttpResponseDecorator, Object)
550             * @see #defaultFailureHandler(HttpResponseDecorator)
551             * @return the default response handler map.
552             */
553            protected Map<Object,Closure> buildDefaultResponseHandlers() {
554                    Map<Object,Closure> map = new StringHashMap<Closure>();
555                    map.put( Status.SUCCESS, 
556                                    new MethodClosure(this,"defaultSuccessHandler"));
557                    map.put(  Status.FAILURE,
558                                    new MethodClosure(this,"defaultFailureHandler"));
559                    
560                    return map;
561            }

563            /**
564             * <p>This is the default <code>response.success</code> handler.  It will be 
565             * executed if the response is not handled by a status-code-specific handler 
566             * (i.e. <code>response.'200'= {..}</code>) and no generic 'success' handler 
567             * is given (i.e. <code>response.success = {..}</code>.)  This handler simply 
568             * returns the parsed data from the response body.  In most cases you will 
569             * probably want to define a <code>response.success = {...}</code> handler 
570             * from the request closure, which will replace the response handler defined 
571             * by this method.  </p>
572             * 
573             * <h4>Note for parsers that return streaming content:</h4>
574             * <p>For responses parsed as {@link ParserRegistry#parseStream(HttpResponse) 
575             * BINARY} or {@link ParserRegistry#parseText(HttpResponse) TEXT}, the 
576             * parser will return streaming content -- an <code>InputStream</code> or 
577             * <code>Reader</code>.  In these cases, this handler will buffer the the 
578             * response content before the network connection is closed.  </p>
579             * 
580             * <p>In practice, a user-supplied response handler closure is 
581             * <i>designed</i> to handle streaming content so it can be read directly from 
582             * the response stream without buffering, which will be much more efficient.
583             * Therefore, it is recommended that request method variants be used which 
584             * explicitly accept a response handler closure in these cases.</p>
585             *  
586             * @param resp HTTP response
587             * @param parsedData parsed data as resolved from this instance's {@link ParserRegistry}
588             * @return the parsed data object (whatever the parser returns).
589             * @throws ResponseParseException if there is an error buffering a streaming
590             *   response.
591             */
592            protected Object defaultSuccessHandler( HttpResponseDecorator resp, Object parsedData ) 
593                            throws ResponseParseException {
594                    try {
595                            //If response is streaming, buffer it in a byte array:
596                            if ( parsedData instanceof InputStream ) {
597                                    ByteArrayOutputStream buffer = new ByteArrayOutputStream();
598                                    DefaultGroovyMethods.leftShift( buffer, (InputStream)parsedData );
599                                    parsedData = new ByteArrayInputStream( buffer.toByteArray() );
600                            }
601                            else if ( parsedData instanceof Reader ) {
602                                    StringWriter buffer = new StringWriter();
603                                    DefaultGroovyMethods.leftShift( buffer, (Reader)parsedData );
604                                    parsedData = new StringReader( buffer.toString() );
605                            }
606                            else if ( parsedData instanceof Closeable )
607                                    log.warn( "Parsed data is streaming, but will be accessible after " +
608                                                    "the network connection is closed.  Use at your own risk!" );
609                            return parsedData;
610                    }
611                    catch ( IOException ex ) {
612                            throw new ResponseParseException( resp, ex );
613                    }
614            }

879                    public RequestConfigDelegate( HttpRequestBase request, Object contentType, 
880                                    Map<?,?> defaultRequestHeaders,

882                            if ( request == null ) throw new IllegalArgumentException( 
883                                            "Internal error - HttpRequest instance cannot be null" );

888                                    this.requestContentType = defaultRequestContentType.toString();
889                            this.responseHandlers.putAll( defaultResponseHandlers );
890                            URI uri = request.getURI();
891                            if ( uri != null ) this.uri = new URIBuilder(uri);
892                    }

1008                     * Valid arguments:
1009                     * <dl>
1010                     *   <dt>uri</dt><dd>Either a URI, URL, or object whose 
1011                     *      <code>toString()</code> method produces a valid URI string. 
1012                     *      If this parameter is not supplied, the HTTPBuilder's default 
1013                     *      URI is used.</dd>
1014                     *   <dt>path</dt><dd>Request path that is merged with the URI</dd>
1015                     *   <dt>query</dt><dd>Map of URL query parameters</dd>
1016                     *   <dt>headers</dt><dd>Map of HTTP headers</dd>
1017                     *   <dt>contentType</dt><dd>Request content type and Accept header.  
1018                     *      If not supplied, the HTTPBuilder's default content-type is used.</dd>
1019                     *   <dt>requestContentType</dt><dd>content type for the request, if it
1020                     *      is different from the expected response content-type</dd>
1021                     *   <dt>body</dt><dd>Request body that will be encoded based on the given contentType</dd>
1022                     * </dl>
1023                     * @param args named parameters to set properties on this delegate.
1024                     * @throws URISyntaxException if the uri argument does not represent a valid URI
1025                     */
1026                    @SuppressWarnings("unchecked")
1027                    protected void setPropertiesFromMap( Map<String,?> args ) throws URISyntaxException {
1028                            if ( args == null ) return;
1029                            if ( args.get( "url" ) != null ) throw new IllegalArgumentException(
1030                                            "The 'url' parameter is deprecated; use 'uri' instead" );
1031                            Object uri = args.get( "uri" );
1032                            if ( uri == null ) uri = defaultURI;
1033                            if ( uri == null ) throw new IllegalStateException( 
1034                                            "Default URI is null, and no 'uri' parameter was given" );
1035                            this.uri = new URIBuilder( convertToURI( uri ) );
1036                            
1037                            Map query = (Map)args.get( "params" );
1038                            if ( query != null ) { 
1039                                    log.warn( "'params' argument is deprecated; use 'query' instead." );
1040                                    this.uri.setQuery( query );
1041                            }
1042                            query = (Map)args.get( "query" );
1043                            if ( query != null ) this.uri.setQuery( query );
1044                            Map headers = (Map)args.get( "headers" );
1045                            if ( headers != null ) this.getHeaders().putAll( headers );
1046                            
1047                            Object path = args.get( "path" );
1048                            if ( path != null ) this.uri.setPath( path.toString() );
1049                            
1050                            Object contentType = args.get( "contentType" );
1051                            if ( contentType != null ) this.setContentType( contentType );
1052                            
1053                            contentType = args.get( "requestContentType" );
1054                            if ( contentType != null ) this.setRequestContentType( contentType.toString() );
1055                            
1056                            Object body = args.get("body");
1057                            if ( body != null ) this.setBody( body );
1058                    }

1094                     * <p>Example:

1099                     *      HTTPBuilder's default EncoderRegistry implementation * /

1101                     *     a : ['one','two','three']

1104                     *   // response content-type is what was specified in the outer request() argument:
1105                     *   response.success = { resp, html -> 
1106                     *   
1107                     *   }
1108                     * }
1109                     * </pre>
1110                     * The <code>send</code> call is equivalent to the following:
1111                     * <pre>
1112                     *   requestContentType = 'text/javascript'
1113                     *   body = { a : ['one','two','three'] }
1114                     * </pre>

1127                     * the associated {@link EncoderRegistry request encoder}.  That is, 
1128                     * the value of <code>body</code> will be interpreted by the encoder
1129                     * associated with the current {@link #getRequestContentType() request 
1130                     * content-type}.

1139                            HttpEntity entity = (HttpEntity)encoder.call( body );
1140                            
1141                            ((HttpEntityEnclosingRequest)this.request).setEntity( entity );
1142                    }
